import { Meta, Story, Canvas, ArgsTable, Source } from '@storybook/addon-docs';
import dedent from 'string-dedent';
import { ButtonPrimary } from '../..';
import { useCopyToClipboard } from '../useCopyToClipboard';

export const UseCopyToClipboardExample = () => {
  const [copyText, copyState] = useCopyToClipboard('Example text to copy', '.copy-target', {
    copyable: 'Copy Sample Text'
  });
  return (
    <div className="hook-story">
      <ButtonPrimary className="copy-target" disabled={copyState === 'copied'}>
        {copyText}
      </ButtonPrimary>
    </div>
  );
};
